#Recebendo o input de um numero inteiro para mostrar os proximos cinco numeros impares

n = int(input())

#-2

#Criando uma variavel para a contagem de numeros impares ja encontrados

counter = 0

#-3

#Comprovando se o numero recebido é par

#Caso ele seja par aumento ele em um para conseguir o proximo impar

if n%2 == 0:
n += 1
print(n)
counter += 1

#-4

#Caso ele seja impar aumento o counter em um e mostro o numero

else:
counter += 1 print(n)

#-5

#Aqui mostro todos os proximos numeros impares ate completar 5

#Se recebi um numero impar ele calculara so mais 4

while counter < 6:
n += 2
counter += 1
print(n)
